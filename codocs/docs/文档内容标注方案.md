Document Annotation Feature Implementation Plan
A feature allowing users to select text in documents, add annotations/comments, mention other users, and collaborate through replies.

User Review Required
IMPORTANT

位置追踪策略: 当文档被编辑时，标注位置会发生变化。推荐使用 内容锚点 + 模糊匹配 方案，因为更可靠且不依赖实时协同。

WARNING

数据库变更: 需要创建 2 个新表 (document_annotations, annotation_replies)。

-- 标注表
CREATE TABLE document_annotations (
    id INT PRIMARY KEY AUTO_INCREMENT,
    document_uuid VARCHAR(36) NOT NULL,
    
    -- 位置信息（内容锚点策略）
    selected_text TEXT NOT NULL,           -- 选中的文本
    context_before VARCHAR(100),           -- 选区前的上下文（用于定位）
    context_after VARCHAR(100),            -- 选区后的上下文
    position_hint INT,                     -- 大致位置提示
    
    -- 标注内容
    content TEXT NOT NULL,
    mentioned_users JSON,                  -- @提及的用户ID数组
    
    -- 状态
    status ENUM('active', 'resolved', 'deleted') DEFAULT 'active',
    
    -- 元数据
    author_id VARCHAR(50) NOT NULL,
    author_name VARCHAR(100),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    resolved_at DATETIME NULL,
    resolved_by VARCHAR(50) NULL,
    deleted_at DATETIME NULL,
    deleted_by VARCHAR(50) NULL,
    
    INDEX idx_document (document_uuid),
    INDEX idx_status (status)
);
-- 回复表
CREATE TABLE annotation_replies (
    id INT PRIMARY KEY AUTO_INCREMENT,
    annotation_id INT NOT NULL,
    content TEXT NOT NULL,
    mentioned_users JSON,
    author_id VARCHAR(50) NOT NULL,
    author_name VARCHAR(100),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    deleted_at DATETIME NULL,
    
    FOREIGN KEY (annotation_id) REFERENCES document_annotations(id) ON DELETE CASCADE
);

Phase 3: Position Tracking
内容锚点算法
interface AnnotationPosition {
    selectedText: string      // 选中文本
    contextBefore: string     // 前 100 字符
    contextAfter: string      // 后 100 字符
    positionHint: number      // 大致位置
}
function findAnnotationPosition(doc: string, anchor: AnnotationPosition): { from: number, to: number } | null {
    // 1. 精确匹配：上下文 + 选中文本
    const pattern = anchor.contextBefore + anchor.selectedText + anchor.contextAfter
    let index = doc.indexOf(pattern)
    if (index !== -1) {
        return { from: index + anchor.contextBefore.length, to: index + anchor.contextBefore.length + anchor.selectedText.length }
    }
    
    // 2. 模糊匹配：仅选中文本 + 位置提示
    const allMatches = findAllOccurrences(doc, anchor.selectedText)
    if (allMatches.length === 1) return allMatches[0]
    
    // 3. 使用位置提示选择最近的匹配
    return findClosestMatch(allMatches, anchor.positionHint)
}
